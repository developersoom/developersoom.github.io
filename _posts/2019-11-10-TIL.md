## useState 구현(미션1)

```js
const React = (()=>{
    let _val
    return {
        render(Component) {
            const comp = Component();
            comp.render();
            return comp;
        },
        useState(initVal) {
            const state = _val || initVal
            const setState = newVal => {
                _val = newVal
            }
            return [state, setState]
        }
    }

}
)();

function MyComponent() {
    const [foo,setFoo] = React.useState(10);
    return {
        render() {
            setFoo(foo + 1);
            console.log(`foo value is ${foo}`);
        }
    }
}

React.render(MyComponent);
React.render(MyComponent);
React.render(MyComponent);

//실행결과 
// foo value is 10
// foo value is 11
// foo value is 12

```

  let _val;
